# 首次安装

```bash
sudo apt-get install -y libzbar-dev fonts-wqy-zenhei
pip3 install openai==1.35.9 requests httpx==0.27.2 psutil flask
sudo dpkg -i src/origincar/3rdparty/aurora930/ros-humble-deptrum-ros-driver-aurora930_0.2.4-0jammy_arm64.deb
```

# 编译

```bash
cd /userdata/dev_ws
colcon build
source install/setup.bash
```

# 启动 Aurora RGB

```bash
ros2 launch deptrum-ros-driver-aurora930 aurora930_launch.py
ros2 topic hz /rgb/image_raw
```

# 启动相机、感知、底盘和大模型

```bash
ros2 launch origincar_bringup usb_websocket_display.launch.py \
  aurora_rgb_topic:=/rgb/image_raw \
  aurora_detect_topic:=/aurora/person_detection
```

# 启动控制

```bash
ros2 launch control control.launch.py \
  v_line:=0.8 \
  kp_line:=0.006 \
  v_avoid:=0.8 \
  kp_avoid:=0.0035 \
  y_zt:=157 \
  y_p:=435 \
  y_qrcode:=167 \
  y_line:=200
```

执行以上命令后小车保持停车。等待所有节点启动完成后，在另一个终端发车：

```bash
cd /userdata/dev_ws
source install/setup.bash
ros2 topic pub --once /start std_msgs/msg/Int32 "{data: 1}"
```

# 启动结果显示屏

```bash
ros2 run display display
```
